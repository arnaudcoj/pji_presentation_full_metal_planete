while (window.isOpen())
{
    //Handle events
    sf::Event event;
    while (window.pollEvent(event)) {
        if (event.type == sf::Event::EventType::Closed)
            window.close();
    }

    //Update scene
    objects.uptdate();

    //Render cycle
    window.clear();
    objects.draw();
    window.display();
}
